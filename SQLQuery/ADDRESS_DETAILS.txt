CREATE TABLE ADDRESS_DETAILS( 
	USER_ID VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	STREET_NO VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	ZIP_CODE VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT ADDRESS_DETAILS_PK PRIMARY KEY (USER_ID),
       CONSTRAINT ADDRESS_FK FOREIGN KEY (ZIP_CODE) REFERENCES AREA_DETAILS(ZIP_CODE)  ON DELETE SET NULL);
